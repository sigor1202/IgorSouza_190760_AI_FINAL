tree("Root")
	parallel
		repeat mute tree("Patrol")
		repeat mute tree("Dead")

tree("Patrol")
	fallback
		tree("Seek")
		while IsHealthLessThan(40.0) 
			fallback
				tree("Wander")

tree("Seek")
	while SeePlayer
		sequence
			seek
			Wait(0.2)
			Fire
			
			

tree("Dead")
	sequence
		IsHealthLessThan(0.1)
		Explode

tree("Wander")
	sequence
		PickRandomDestination
		MoveToDestination
